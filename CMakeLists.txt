cmake_minimum_required(VERSION 3.16)

project(SQLyogCommunity LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(QT NAMES Qt6 Qt5 REQUIRED COMPONENTS Core Gui Widgets Network LinguistTools)
find_package(Qt${QT_VERSION_MAJOR} REQUIRED COMPONENTS Core Gui Widgets Network LinguistTools)

set(TS_FILES
    localization/app_en.ts
)
qt_add_translation(QM_FILES ${TS_FILES})

file(GLOB_RECURSE SOURCES src/*.cpp)
qt_add_resources(APP_RESOURCES resources/resources.qrc)

qt_add_executable(${PROJECT_NAME}
    ${SOURCES}
    ${APP_RESOURCES}
    ${QM_FILES}
)

target_link_libraries(${PROJECT_NAME} PRIVATE
    Qt${QT_VERSION_MAJOR}::Core
    Qt${QT_VERSION_MAJOR}::Gui
    Qt${QT_VERSION_MAJOR}::Widgets
    Qt${QT_VERSION_MAJOR}::Network
)
